CREATE VIEW [dbo].[V_PROFITABILITY] AS
select BRANCH_ID,SUM(tran_amt) AS TRANSACTION_AMOUNT,SUM(tran_fee_amt) AS FEE_AMOUNT, SUM(TRAN_AMT+TRAN_FEE_AMT) AS TOTAL_PROFITABILITY from [dbo].[tbl_transactions_fact] 
GROUP BY BRANCH_ID
GO
